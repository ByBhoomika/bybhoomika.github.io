import numpy as np import matplotlib.pyplot as plt # ========================================== # AERODYNAMIC DRAG SIMULATION - PORTFOLIO # ========================================== # Constants rho = 1.225 # Air density (kg/m^3) A = 1.0 # Frontal area (m^2) # Speed range (0 to 50 m/s) speeds = np.linspace(0, 50, 300) # Drag coefficients shapes = { "Cube": 1.05, "Sphere": 0.47, "Sedan Car": 0.28, "Streamlined Body": 0.15 } # Create graph plt.figure(figsize=(12,7)) # Plot each line for shape, Cd in shapes.items(): drag_force = 0.5 * rho * speeds**2 * Cd * A plt.plot(speeds, drag_force, linewidth=3, label=shape) # Styling plt.title("Aerodynamic Drag Force vs Speed", fontsize=20, fontweight='bold') plt.xlabel("Speed (m/s)", fontsize=14) plt.ylabel("Drag Force (N)", fontsize=14) plt.xticks(fontsize=12) plt.yticks(fontsize=12) plt.grid(True, linestyle='--', alpha=0.6) plt.legend(fontsize=12, loc='upper left') plt.tight_layout() # Save graph image plt.savefig("aerodynamicdrag.png", dpi=300) # Show graph plt.show()
